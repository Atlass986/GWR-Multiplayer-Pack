country_event = {
	id = FRA_belgium_puppet_offer.1
	title = FRA_belgium_puppet_offer.1.t
	desc = FRA_belgium_puppet_offer.1.d
	picture = GFX_France_Demand

	is_triggered_only = yes

	option = { # Accept
		name = FRA_benelux_puppet_offer.1.a
		ai_chance = {
			base = 10
			modifier = {
				factor = 2
				has_opinion = {
					target = FRA
					value > 50
				}
			}
			modifier = {
				factor = 0.5
				has_war = yes
			}
		}
		FRA = {
			puppet = ROOT
		}
		FRA = {
			country_event = {
				id = FRA_benelux_puppet_offer_accept.1
				days = 1
			}
		}
	}

	option = { # Refuse
		name = FRA_benelux_puppet_offer.1.b
		ai_chance = {
			base = 10
			modifier = {
				factor = 2
				has_opinion = {
					target = FRA
					value < 0
				}
			}
		}
		FRA = {
			add_opinion_modifier = {
				target = ROOT
				modifier = refused_puppet_offer
			}
		}
		FRA = {
			country_event = {
				id = FRA_belgium_puppet_offer.3
				days = 1
			}
		}
	}
}

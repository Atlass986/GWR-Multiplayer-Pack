 	pay_civ_factory = {
		visible = { THIS = { NOT = { has_war_with = ROOT } } }
		selectable = {  
			ROOT = {
				is_subject = no
				any_owned_state = { is_controlled_by = PREV industrial_complex > 0}
			}
			THIS = {
				any_owned_state = { 
					free_building_slots = {
						building = industrial_complex
						size > 0
					}
				}

			}
		}

		requires_acceptance = no
		cost = 5
		command_power = 0

		show_acceptance_on_action_button = no

		icon = 4 

		on_sent_effect = {}

		complete_effect = {
			ROOT = {	
				random_owned_controlled_state = { limit = { industrial_complex > 0 } remove_building = { type = industrial_complex level = 1 } }
			}
			THIS = {	
				random_owned_controlled_state = {
					limit = {
						free_building_slots = {
							building = industrial_complex
							size > 0
						}
					}
					add_building_construction = {
						type = industrial_complex
						level = 1
						instant_build = yes
					}
				}
				
			}
			log = "Civ Factory: [?ROOT.GetName] -> [?THIS.GetName]"
		}

		reject_effect = {}

		send_description = pay_civ_factory_send_text

		receive_description =pay_civ_get_message

		accept_title =pay_civ_get_title

		accept_description =pay_civ_get_message

		reject_title =pay_civ_get_title

		reject_description =pay_civ_get_message

		ai_acceptance = {
			condition = { base = 0 }
		}
